(function () { const API = window.AIStationAPI; const loginScreen = document.getElementById("loginScreen"); const appLayout = document.getElementById("appLayout"); const loginBtn = document.getElementById("loginBtn"); const wsGrid = document.getElementById("wsGrid"); const refreshDash = document.getElementById("refreshDash"); let adminToken = ""; let currentAdmin = null; function formatMoney(value) { const amount = Number(value || 0); } return []; } function renderRows(tbody, rowsHtml, colspan) { if (!tbody) return; tbody.innerHTML = rowsHtml || `No data found`; } function renderDashboardStats(dashboard, transactions, contacts) { const activeSessions = Number(dashboard?.activeSessions) || safeArray(dashboard, ["sessions", "active"]).length || 0; } function renderRecentTransactions(transactions) { const items = safeArray(transactions, ["transactions"]).slice(0, 10); const html = items .map((item) => { return ` ${item.phone || item.customerPhone || "—"} ${item.plan || item.planName || "—"} ${formatMoney(item.amount)} ${item.status || "—"} ${formatDate(item.createdAt || item.date || item.timestamp)} `; }) .join(""); renderRows(sessBody, html, 6); } function renderContacts(contacts) { const items = safeArray(contacts, ["contacts", "messages"]); const html = items .map((item) => { return ` ${item.name || "—"} ${item.email || "—"} ${item.phone || "—"} ${item.interest || "—"} ${item.message || "—"} ${formatDate(item.createdAt || item.date)} `; }) .join(""); renderRows(contactBody, html, 6); } function renderWorkstations(data) { const items = safeArray(data, ["workstations"]) || []; if (!wsGrid) return; if (!items.length) { wsGrid.innerHTML = `

No workstations found

`; return; } wsGrid.innerHTML = items .map((item) => { const status = (item.status || "idle").toLowerCase(); return `

${item.workstationId || item.id || "WS"}

${status}

Current user: ${item.phone || item.customerPhone || "—"}

Minutes left: ${item.minutesLeft ?? item.balanceMinutes ?? "—"}

Updated: ${formatDate(item.updatedAt || item.lastSeen)}

`; }) .join(""); } async function loadAdminData() { try { const [dashboard, transactions, contacts, workstations] = await Promise.all([ API.getAdminDashboard(), API.getAdminTransactions(), API.getContacts(), API.getWorkstations(), ]); renderDashboardStats(dashboard, transactions, contacts); renderRecentTransactions(transactions); renderAllTransactions(transactions); renderSessions(dashboard); renderContacts(contacts); renderWorkstations(workstations); } catch (error) { setText(loginAlert, error.message || "Failed to load admin data"); } } async function handleLogin() { const email = emailInput?.value?.trim(); const password = passwordInput?.value || ""; if (!email || !password) { setText(loginAlert, "Enter email and password"); return; } loginBtn.disabled = true; setText(loginAlert, "Signing in..."); try { const result = await API.login(email, password); adminToken = result?.token || result?.accessToken || result?.data?.token || ""; currentAdmin = result?.admin || result?.user || result?.data?.admin || null; if (!adminToken) { throw new Error("Login succeeded but token was not returned"); } API.setToken(adminToken); setText(loginAlert, ""); setLoginState(true); activatePage("dashboard"); await loadAdminData(); } catch (error) { setText(loginAlert, error.message || "Login failed"); setLoginState(false); } finally { loginBtn.disabled = false; } } function handleLogout() { adminToken = ""; currentAdmin = null; API.setToken(""); setLoginState(false); setText(loginAlert, ""); if (emailInput) emailInput.value = ""; if (passwordInput) passwordInput.value = ""; } loginBtn?.addEventListener("click", handleLogin); logoutBtn?.addEventListener("click", handleLogout); refreshDash?.addEventListener("click", loadAdminData); passwordInput?.addEventListener("keydown", (event) => { if (event.key === "Enter") { handleLogin(); } }); navItems.forEach((item) => { item.addEventListener("click", () => { activatePage(item.dataset.page); }); }); setLoginState(false); })();
