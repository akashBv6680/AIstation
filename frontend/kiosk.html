(function () { const API = window.AIStationAPI; const customerPhone = document.getElementById("customerPhone"); const customerName = document.getElementById("customerName"); const checkBalanceBtn = document.getElementById("checkBalanceBtn"); const balanceStatus = document.getElementById("balanceStatus"); const balanceMinutes = document.getElementById("balanceMinutes"); const balanceMeta = document.getElementById("balanceMeta"); const planCards = Array.from(document.querySelectorAll(".plan-card")); const selectedPlanName = document.getElementById("selectedPlanName"); const selectedPlanMeta = document.getElementById("selectedPlanMeta"); const buyCreditsBtn = document.getElementById("buyCreditsBtn"); const paymentStatus = document.getElementById("paymentStatus"); const workstationId = document.getElementById("workstationId"); const sessionMinutes = document.getElementById("sessionMinutes"); const startSessionBtn = document.getElementById("startSessionBtn"); const endSessionBtn = document.getElementById("endSessionBtn"); const sessionStatus = document.getElementById("sessionStatus"); const sessionLiveState = document.getElementById("sessionLiveState"); const sessionMeta = document.getElementById("sessionMeta"); const loadHistoryBtn = document.getElementById("loadHistoryBtn"); const historyBody = document.getElementById("historyBody"); let activePlan = { plan: "starter", minutes: 100, amount: 150, }; let activeSessionId = ""; function setText(el, value) { if (el) el.textContent = value; } function normalizePhone(value) { return String(value || "").replace(/\D/g, "").slice(0, 10); } function formatMoney(value) { const amount = Number(value || 0); return new Intl.NumberFormat("en-IN", { style: "currency", currency: "INR", maximumFractionDigits: 0, }).format(amount); } function formatDate(value) { if (!value) return "—"; const d = new Date(value); if (Number.isNaN(d.getTime())) return value; return d.toLocaleString("en-IN", { dateStyle: "medium", timeStyle: "short", }); } function safeArray(data, keys = []) { if (Array.isArray(data)) return data; for (const key of keys) { if (Array.isArray(data?.[key])) return data[key]; } return []; } function getPhone() { return normalizePhone(customerPhone?.value); } function validatePhone() { const phone = getPhone(); if (phone.length !== 10) { throw new Error("Enter a valid 10-digit phone number"); } return phone; } function updatePlanUI() { setText( selectedPlanName, activePlan.plan === "pro" ? "Pro Pass" : activePlan.plan === "explorer" ? "Explorer" : "Starter" ); setText( selectedPlanMeta, `${activePlan.minutes} mins • ${formatMoney(activePlan.amount)}` ); } function setSelectedPlan(card) { planCards.forEach((item) => item.classList.remove("active")); card.classList.add("active"); activePlan = { plan: card.dataset.plan || "starter", minutes: Number(card.dataset.minutes || 0), amount: Number(card.dataset.amount || 0), }; updatePlanUI(); } async function checkBalance() { try { const phone = validatePhone(); checkBalanceBtn.disabled = true; setText(balanceStatus, "Checking balance..."); const result = await API.getBalance(phone); const minutes = Number(result?.balanceMinutes) || Number(result?.minutes) || Number(result?.credits) || 0; setText(balanceMinutes, `${minutes} mins`); setText( balanceMeta, result?.message || `Balance loaded for ${phone}` ); setText(balanceStatus, "Balance loaded"); } catch (error) { setText(balanceStatus, error.message || "Failed to check balance"); setText(balanceMinutes, "0 mins"); setText(balanceMeta, "Unable to fetch balance"); } finally { checkBalanceBtn.disabled = false; } } async function buyCredits() { try { const phone = validatePhone(); const name = customerName?.value?.trim() || ""; buyCreditsBtn.disabled = true; setText(paymentStatus, "Creating payment..."); const paymentPayload = { phone, name, plan: activePlan.plan, minutes: activePlan.minutes, amount: activePlan.amount, }; const createResult = await API.createPayment(paymentPayload); if ( createResult?.status === "success" || createResult?.success === true || createResult?.paymentId || createResult?.orderId ) { setText( paymentStatus, `Payment created for ${formatMoney(activePlan.amount)}. Credits will reflect after verification.` ); } else { setText( paymentStatus, createResult?.message || "Payment request created" ); } await checkBalance(); } catch (error) { setText(paymentStatus, error.message || "Failed to create payment"); } finally { buyCreditsBtn.disabled = false; } } async function startSession() { try { const phone = validatePhone(); const ws = workstationId?.value; const mins = Number(sessionMinutes?.value || 0); if (!ws) { throw new Error("Choose a workstation"); } if (!mins || mins < 10) { throw new Error("Enter at least 10 minutes"); } startSessionBtn.disabled = true; setText(sessionStatus, "Starting session..."); const payload = { phone, workstationId: ws, minutes: mins, }; const result = await API.startSession(payload); activeSessionId = result?.sessionId || result?.id || result?.data?.sessionId || ""; setText(sessionLiveState, "Session active"); setText( sessionMeta, `${ws} • ${mins} mins • ${phone}` ); setText( sessionStatus, result?.message || "Session started successfully" ); await checkBalance(); await loadHistory(); } catch (error) { setText(sessionStatus, error.message || "Failed to start session"); } finally { startSessionBtn.disabled = false; } } async function endSession() { try { const phone = validatePhone(); endSessionBtn.disabled = true; setText(sessionStatus, "Ending session..."); const payload = activeSessionId ? { sessionId: activeSessionId, phone } : { phone, workstationId: workstationId?.value || "", }; const result = await API.endSession(payload); setText(sessionLiveState, "No active session"); setText(sessionMeta, "Choose a workstation and start"); setText( sessionStatus, result?.message || "Session ended successfully" ); activeSessionId = ""; await checkBalance(); await loadHistory(); } catch (error) { setText(sessionStatus, error.message || "Failed to end session"); } finally { endSessionBtn.disabled = false; } } async function loadHistory() { try { const phone = validatePhone(); loadHistoryBtn.disabled = true; const result = await API.getSessions(phone); const items = safeArray(result, ["sessions", "data"]); if (!items.length) { historyBody.innerHTML = `No sessions found`; return; } historyBody.innerHTML = items .map((item) => { const minutes = item.minutes || item.durationMinutes || item.duration || 0; return ` ${item.sessionId || item.id || "—"} ${item.workstationId || item.workstation || "—"} ${minutes} ${item.status || "—"} ${formatDate(item.startedAt || item.createdAt)} `; }) .join(""); } catch (error) { historyBody.innerHTML = `${error.message || "Failed to load history"}`; } finally { loadHistoryBtn.disabled = false; } } planCards.forEach((card) => { card.addEventListener("click", () => setSelectedPlan(card)); }); checkBalanceBtn?.addEventListener("click", checkBalance); buyCreditsBtn?.addEventListener("click", buyCredits); startSessionBtn?.addEventListener("click", startSession); endSessionBtn?.addEventListener("click", endSession); loadHistoryBtn?.addEventListener("click", loadHistory); customerPhone?.addEventListener("input", () => { customerPhone.value = normalizePhone(customerPhone.value); }); updatePlanUI(); })();
